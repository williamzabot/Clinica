
    create table tb_doctors (
       id bigint not null auto_increment,
        city varchar(255),
        complement varchar(255),
        neighborhood varchar(255),
        number varchar(255),
        state varchar(255),
        street varchar(255),
        zip_code varchar(255),
        crm varchar(255),
        email varchar(255),
        name varchar(255),
        via_health_insurance bit not null,
        specialty_id bigint,
        primary key (id)
    ) engine=InnoDB;

    create table tb_employees (
       id bigint not null auto_increment,
        cpf varchar(255),
        email varchar(255),
        name varchar(255),
        address_id bigint,
        primary key (id)
    ) engine=InnoDB;

    create table tb_patients (
       id bigint not null auto_increment,
        cpf varchar(255),
        email varchar(255),
        name varchar(255),
        register_date date,
        address_id bigint,
        primary key (id)
    ) engine=InnoDB;

    create table tb_specialties (
       id bigint not null auto_increment,
        description TEXT,
        name varchar(255),
        primary key (id)
    ) engine=InnoDB;
